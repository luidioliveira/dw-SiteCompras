CREATE TABLE CONTATO
(
    ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
    NOME VARCHAR(50),
    IDADE INTEGER
);

CREATE TABLE CATEGORIA
(
    ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1), 
    NMCATEGORIA VARCHAR(50)
);

CREATE TABLE PRODUTO
(
    ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
    IDCATEGORIA INT, 
    NOME VARCHAR(50), 
    DESCRICAO VARCHAR(50),
    VALOR DOUBLE
);

CREATE TABLE ADMINISTRADOR
(
    ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
    LOGIN VARCHAR(10),
    SENHA VARCHAR(10)
);

CREATE TABLE CLIENTE
(
    ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
    NOME VARCHAR(50), 
    ENDERECO VARCHAR(50),
    BAIRRO VARCHAR(50),
    CIDADE VARCHAR(50),
    CEP VARCHAR(50),
    ESTADO VARCHAR(2),
    REFERENCIA VARCHAR(50),
    CPF VARCHAR(20),
    RI VARCHAR(50),
    TELFIXO VARCHAR(50),
    CELULAR VARCHAR(50),
    CARTAOCRED VARCHAR(50),
    BANDEIRA VARCHAR(50)
);

CREATE TABLE COMPRA
(
    ID INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
    IDCLIENTE INT,
    IDPRODUTO INT
)
